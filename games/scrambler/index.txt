        // Game State
        const state = {
            sentenceWords: [],
            answerWords: [],
            sentenceOrder: [],
            answerOrder: [],
            colorMode: true,
            easyMode: true,
            moveCount: 0,
            draggedElement: null,
            dragOffset: { x: 0, y: 0 }
        };

        // Audio Context for sounds
        let audioCtx = null;

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
        }

        function playSound(frequency, duration, type = 'sine') {
            if (!audioCtx) return;
            const oscillator = audioCtx.createOscillator();
            const gainNode = audioCtx.createGain();
            oscillator.connect(gainNode);
            gainNode.connect(audioCtx.destination);
            oscillator.frequency.value = frequency;
            oscillator.type = type;
            gainNode.gain.setValueAtTime(0.3, audioCtx.currentTime);
            gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
            oscillator.start(audioCtx.currentTime);
            oscillator.stop(audioCtx.currentTime + duration);
        }

        function playSuccessSound() {
            playSound(523.25, 0.1);
            setTimeout(() => playSound(659.25, 0.1), 100);
            setTimeout(() => playSound(783.99, 0.2), 200);
        }

        function playErrorSound() {
            playSound(200, 0.15, 'square');
            setTimeout(() => playSound(150, 0.25, 'square'), 150);
        }

        // Split text into words
        function splitIntoWords(text) {
            return text.trim().split(/\s+/).filter(w => w.length > 0);
        }

        // Shuffle array
        function shuffle(array) {
            const arr = [...array];
            for (let i = arr.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [arr[i], arr[j]] = [arr[j], arr[i]];
            }
            return arr;
        }

        // Create fragment element
        function createFragment(word, index, isSentence) {
            const frag = document.createElement('div');
            frag.className = `fragment ${isSentence ? 'sentence-fragment' : 'answer-fragment'}`;
            frag.textContent = word;
            frag.dataset.word = word;
            frag.dataset.index = index;
            frag.dataset.type = isSentence ? 'sentence' : 'answer';
            frag.dataset.isPlaced = 'false';

            // Mouse events
            frag.addEventListener('mousedown', handleDragStart);

            // Touch events
            frag.addEventListener('touchstart', handleTouchStart, { passive: false });

            return frag;
        }

        // Magnetic snap distance
        const MAGNET_DISTANCE = 60;

        // Get all fragments in a zone
        function getFragmentsInZone(zoneId) {
            const zone = document.getElementById(zoneId);
            return Array.from(zone.querySelectorAll('.fragment'));
        }

        // Find nearest fragment in same zone
        function findNearestFragment(element, clientX, clientY) {
            const type = element.dataset.type;
            const zoneId = type === 'sentence' ? 'sentence-zone' : 'answer-zone';
            const fragments = getFragmentsInZone(zoneId).filter(f => f !== element);

            if (fragments.length === 0) return null;

            let nearest = null;
            let nearestDist = Infinity;

            const elRect = {
                left: clientX - 50,
                top: clientY - 25,
                right: clientX + 50,
                bottom: clientY + 25,
                centerX: clientX,
                centerY: clientY
            };

            fragments.forEach(f => {
                const rect = f.getBoundingClientRect();
                const centerX = rect.left + rect.width / 2;
                const centerY = rect.top + rect.height / 2;
                const dist = Math.hypot(centerX - clientX, centerY - clientY);

                if (dist < nearestDist && dist < MAGNET_DISTANCE * 2) {
                    nearestDist = dist;
                    nearest = { element: f, centerX, centerY };
                }
            });

            return nearest;
        }

        // Snap to position relative to another fragment
        function snapToFragment(element, targetElement, position) {
            const targetRect = targetElement.getBoundingClientRect();
            const zone = element.closest('.zone');
            const zoneRect = zone.getBoundingClientRect();

            let newLeft, newTop;

            if (position === 'before') {
                newLeft = targetRect.left - zoneRect.left - element.offsetWidth - 10;
                newTop = targetRect.top - zoneRect.top;
            } else {
                newLeft = targetRect.right - zoneRect.left + 10;
                newTop = targetRect.top - zoneRect.top;
            }

            element.style.left = newLeft + 'px';
            element.style.top = newTop + 'px';
        }

        // Reorder fragments in a zone based on their positions
        function reorderZone(zoneId) {
            const zone = document.getElementById(zoneId);
            const zoneRect = zone.getBoundingClientRect();
            const fragments = Array.from(zone.querySelectorAll('.fragment'));

            if (fragments.length === 0) return;

            // Sort by horizontal position
            fragments.sort((a, b) => {
                const rectA = a.getBoundingClientRect();
                const rectB = b.getBoundingClientRect();
                return rectA.left - rectB.left;
            });

            // Reposition to be evenly spaced
            const padding = 30;
            const availableWidth = zoneRect.width - padding * 2;
            const totalFragments = fragments.length;
            const spacing = totalFragments > 1 ? availableWidth / (totalFragments - 1) : availableWidth / 2;

            fragments.forEach((frag, i) => {
                const left = padding + (i * spacing) - (frag.offsetWidth / 2) + (spacing / 2);
                const top = (zoneRect.height - frag.offsetHeight) / 2 - 20;
                frag.style.left = Math.max(padding, left) + 'px';
                frag.style.top = Math.max(40, top) + 'px';
            });
        }

        // Drag and Drop Handlers
        function handleDragStart(e) {
            e.preventDefault();
            initAudio();
            const element = e.target.closest('.fragment');
            startDrag(element, e.clientX, e.clientY);
        }

        function handleTouchStart(e) {
            e.preventDefault();
            initAudio();
            const element = e.target.closest('.fragment');
            const touch = e.touches[0];
            startDrag(element, touch.clientX, touch.clientY);
        }

        function startDrag(element, clientX, clientY) {
            state.draggedElement = element;
            element.classList.add('dragging');

            const rect = element.getBoundingClientRect();
            state.dragOffset.x = clientX - rect.left;
            state.dragOffset.y = clientY - rect.top;

            document.addEventListener('mousemove', handleDragMove);
            document.addEventListener('mouseup', handleDragEnd);
            document.addEventListener('touchmove', handleTouchMove, { passive: false });
            document.addEventListener('touchend', handleTouchEnd);
        }

        function handleDragMove(e) {
            if (!state.draggedElement) return;
            e.preventDefault();
            moveDrag(e.clientX, e.clientY);
        }

        function handleTouchMove(e) {
            if (!state.draggedElement) return;
            e.preventDefault();
            const touch = e.touches[0];
            moveDrag(touch.clientX, touch.clientY);
        }

        function moveDrag(clientX, clientY) {
            const el = state.draggedElement;
            const zone = el.closest('.zone');
            const zoneRect = zone.getBoundingClientRect();

            // Allow free movement within reasonable bounds
            let newLeft = clientX - state.dragOffset.x - zoneRect.left;
            let newTop = clientY - state.dragOffset.y - zoneRect.top;

            // Keep within zone bounds with some padding
            const padding = 10;
            newLeft = Math.max(padding, Math.min(newLeft, zoneRect.width - el.offsetWidth - padding));
            newTop = Math.max(padding, Math.min(newTop, zoneRect.height - el.offsetHeight - padding));

            el.style.left = newLeft + 'px';
            el.style.top = newTop + 'px';
        }

        function handleDragEnd(e) {
            endDrag(e.clientX, e.clientY);
        }

        function handleTouchEnd(e) {
            const touch = e.changedTouches[0];
            endDrag(touch.clientX, touch.clientY);
        }

        function endDrag(clientX, clientY) {
            const element = state.draggedElement;
            if (!element) return;

            element.classList.remove('dragging');

            // Find which zone the drop is in
            const zones = document.querySelectorAll('.zone');
            let targetZone = null;

            zones.forEach(zone => {
                const rect = zone.getBoundingClientRect();
                if (clientX >= rect.left && clientX <= rect.right &&
                    clientY >= rect.top && clientY <= rect.bottom) {
                    targetZone = zone;
                }
            });

            if (targetZone) {
                const zoneId = targetZone.id;
                const elementType = element.dataset.type;
                const targetType = zoneId === 'sentence-zone' ? 'sentence' : 'answer';

                // Only allow dropping in matching zone type
                if (elementType !== targetType) {
                    // Wrong zone - just reorder in current zone
                    const currentZone = elementType === 'sentence' ? 'sentence-zone' : 'answer-zone';
                    reorderZone(currentZone);
                    if (state.easyMode) {
                        playErrorSound();
                    }
                    document.removeEventListener('mousemove', handleDragMove);
                    document.removeEventListener('mouseup', handleDragEnd);
                    document.removeEventListener('touchmove', handleTouchMove);
                    document.removeEventListener('touchend', handleTouchEnd);
                    state.draggedElement = null;
                    checkWin();
                    return;
                }

                element.dataset.isPlaced = 'true';
                state.moveCount++;
                updateStats();

                // Check for magnetic snap to other words in the same zone
                const nearest = findNearestFragment(element, clientX, clientY);

                if (nearest) {
                    const elementCenterX = element.getBoundingClientRect().left + element.offsetWidth / 2;
                    const position = elementCenterX < nearest.centerX ? 'before' : 'after';
                    snapToFragment(element, nearest.element, position);
                }

                // Reorder all fragments in the zone
                reorderZone(zoneId);

                if (state.easyMode) {
                    playSuccessSound();
                }
            }

            document.removeEventListener('mousemove', handleDragMove);
            document.removeEventListener('mouseup', handleDragEnd);
            document.removeEventListener('touchmove', handleTouchMove);
            document.removeEventListener('touchend', handleTouchEnd);

            state.draggedElement = null;

            // Check win condition
            checkWin();
        }

        function checkWin() {
            const sentenceZone = document.getElementById('sentence-zone');
            const answerZone = document.getElementById('answer-zone');

            const sentenceFragments = sentenceZone.querySelectorAll('.fragment');
            const answerFragments = answerZone.querySelectorAll('.fragment');

            // Check if all fragments are placed
            if (sentenceFragments.length + answerFragments.length !== state.sentenceWords.length + state.answerWords.length) {
                return;
            }

            // Sort fragments by horizontal position (left to right)
            const sortedSentence = Array.from(sentenceFragments).sort((a, b) => {
                return a.getBoundingClientRect().left - b.getBoundingClientRect().left;
            });

            const sortedAnswer = Array.from(answerFragments).sort((a, b) => {
                return a.getBoundingClientRect().left - b.getBoundingClientRect().left;
            });

            // Verify sentence order
            let sentenceCorrect = true;
            const sentenceArr = sortedSentence.map(f => f.dataset.index);
            for (let i = 0; i < sentenceArr.length; i++) {
                if (parseInt(sentenceArr[i]) !== i) {
                    sentenceCorrect = false;
                    break;
                }
            }

            // Verify answer order
            let answerCorrect = true;
            const answerArr = sortedAnswer.map(f => f.dataset.index);
            for (let i = 0; i < answerArr.length; i++) {
                if (parseInt(answerArr[i]) !== i) {
                    answerCorrect = false;
                    break;
                }
            }

            if (sentenceCorrect && answerCorrect) {
                showWinScreen();
            }
        }

        function showWinScreen() {
            document.getElementById('final-moves').textContent = state.moveCount;
            document.getElementById('game-complete').classList.add('active');
            playSuccessSound();
        }

        function updateStats() {
            document.getElementById('move-count').textContent = state.moveCount;
            const placed = document.querySelectorAll('#sentence-zone .fragment, #answer-zone .fragment').length;
            const total = state.sentenceWords.length + state.answerWords.length;
            document.getElementById('placed-count').textContent = placed;
            document.getElementById('total-count').textContent = total;
        }

        function startGame() {
            const sentence = document.getElementById('target-sentence').value.trim();
            const answer = document.getElementById('target-answer').value.trim();

            if (!sentence || !answer) {
                alert('Please enter both a sentence and an answer.');
                return;
            }

            state.sentenceWords = splitIntoWords(sentence);
            state.answerWords = splitIntoWords(answer);
            state.colorMode = document.getElementById('color-mode').checked;
            state.easyMode = document.getElementById('easy-mode').checked;
            state.moveCount = 0;

            // Apply color mode
            if (!state.colorMode) {
                document.body.classList.add('no-color-mode');
            } else {
                document.body.classList.remove('no-color-mode');
            }

            // Setup game screen
            setupGameScreen();

            // Switch screens
            document.getElementById('config-screen').classList.remove('active');
            document.getElementById('game-screen').classList.add('active');
        }

        function setupGameScreen() {
            const sentenceZone = document.getElementById('sentence-zone');
            const answerZone = document.getElementById('answer-zone');

            // Clear zones but keep the header
            sentenceZone.innerHTML = '<div class="zone-header">📝 Sentence</div>';
            answerZone.innerHTML = '<div class="zone-header">💡 Answer</div>';

            // Shuffle fragments
            const shuffledSentence = shuffle(state.sentenceWords.map((w, i) => ({ word: w, index: i })));
            const shuffledAnswer = shuffle(state.answerWords.map((w, i) => ({ word: w, index: i })));

            // Create sentence fragments - spread horizontally
            shuffledSentence.forEach(({ word, index }) => {
                const frag = createFragment(word, index, true);
                frag.style.position = 'absolute';
                sentenceZone.appendChild(frag);
            });

            // Create answer fragments - spread horizontally
            shuffledAnswer.forEach(({ word, index }) => {
                const frag = createFragment(word, index, false);
                frag.style.position = 'absolute';
                answerZone.appendChild(frag);
            });

            // Initial layout
            setTimeout(() => {
                reorderZone('sentence-zone');
                reorderZone('answer-zone');
            }, 50);

            updateStats();
        }

        function resetGame() {
            setupGameScreen();
            document.getElementById('game-complete').classList.remove('active');
        }

        function playAgain() {
            document.getElementById('game-complete').classList.remove('active');
            resetGame();
        }

        function goBack() {
            document.getElementById('game-complete').classList.remove('active');
            document.getElementById('game-screen').classList.remove('active');
            document.getElementById('config-screen').classList.add('active');
        }

        function toggleFullscreen() {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen().catch(() => {});
            } else {
                document.exitFullscreen();
            }
        }
